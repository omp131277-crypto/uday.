  <script>
    const body = document.body;
    const bulb = document.getElementById('bulb');
    const rope = document.getElementById('rope');
    const handle = document.getElementById('handle');
    let isOn = false;
    let isPulling = false;

    handle.addEventListener('mousedown', () => {
      isPulling = true;
      rope.style.transform = 'translateY(40px)';
      bulb.style.transform = 'translateY(30px)';
      handle.style.transform = 'translateY(40px)';
    });

    document.addEventListener('mouseup', () => {
      if (isPulling) {
        isPulling = false;
        rope.style.transform = 'translateY(0)';
        bulb.style.transform = 'translateY(0)';
        handle.style.transform = 'translateY(0)';

        // Toggle light ON/OFF
        isOn = !isOn;
        if (isOn) {
          bulb.classList.add('glow');
          body.style.backgroundColor = '#fff6b0'; // ON = yellow light
        } else {
          bulb.classList.remove('glow');
          body.style.backgroundColor = '#fff'; // OFF = white
        }
      }
    });
  </script>
